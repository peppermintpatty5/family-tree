\newcommand*{\badgeFemale}[1]{
    \tikz[anchor=base, baseline]
    \node[
        rounded rectangle, draw, very thick,
        text width=6em, minimum height=4ex, align=center,
        white, fill=magenta, double=magenta, very thick
    ] {\tt\large \raisebox{0.3ex}{\female}~#1};
}
\newcommand*{\badgeMale}[1]{
    \tikz[anchor=base, baseline]
    \node[
        signal, draw, signal to=west and east,
        text width=5em, minimum height=4ex, align=center,
        white, fill=blue, double=blue, very thick
    ] {\tt\large \raisebox{-0.1ex}{\male}~#1};
}
\newcommand*{\slotFemale}[1]{
    \tikz[anchor=base, baseline]
    \node[
        rounded rectangle, draw, very thick,
        text width=6em, minimum height=4ex, align=center,
        magenta, very thick, dashed
    ] {\tt\large \raisebox{0.3ex}{\female}~#1};
}
\newcommand*{\slotMale}[1]{
    \tikz[anchor=base, baseline]
    \node[
        signal, draw, signal to=west and east,
        text width=5em, minimum height=4ex, align=center,
        blue, very thick, dashed
    ] {\tt\large \raisebox{-0.1ex}{\male}~#1};
}
